\section{项目定位}
\begin{frame}{一页说清：知序解决什么}
  \begin{columns}[T,onlytextwidth]
    \column{0.55\textwidth}
      \textbf{背景只抓一个问题：}学习资料、任务、专注记录和复盘常分散在不同工具，数据难以形成连续闭环。

      \vspace{0.7em}
      \textbf{项目目标：}在 HarmonyOS 端侧完成“资料整理 -- 任务分解 -- 番茄执行 -- 复盘反馈”，后端同步和远程 AI 作为增强能力。

      \vspace{0.7em}
      \begin{itemize}
        \item 离线状态下，任务、资料、番茄和复盘仍可运行。
        \item 网络可用时，支持 AI 提炼、图片提取任务和云同步。
        \item 答辩重点展示移动端、端侧数据库、原生能力和后端接口的完整链路。
      \end{itemize}
    \column{0.40\textwidth}
      \metric{ArkTS}{前端实现}{ArkUI + Navigation + 原生能力}
      \vspace{0.45em}
      \metric{RDB}{本地优先}{业务数据首先写入端侧数据库}
      \vspace{0.45em}
      \metric{Spring Boot}{后端服务}{REST + MyBatis + JWT + MySQL/H2}
  \end{columns}
\end{frame}

\begin{frame}{前端分层：Index 编排，Store 接业务}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \smallcard{softBlue}{页面层}
        {\code{Index.ets} 维护页面状态、路由栈、Tab 切换和用户事件回调。}
      \vspace{0.22em}
      \smallcard{softGreen}{组件层}
        {\code{TodayView}、\code{TasksView}、\code{ResourcesView}、\code{FocusSessionView}、\code{ReviewView} 只接收状态和回调。}
      \vspace{0.22em}
      \smallcard{softAmber}{服务层}
        {\code{FocusStore} 调用数据库、设置、AI、同步和音频服务，完成后刷新 UI 状态。}
    \column{0.48\textwidth}
      \begin{block}{一次任务新增的链路}
        \small
        \begin{enumerate}
          \item \code{QuickAddTaskCard} 收集标题、备注、截止时间、项目和象限。
          \item \code{Index.ets} 组装 \code{QuickTaskDraft}。
          \item \code{FocusStore.addTaskFromDraft} 规范化字段。
          \item \code{FocusDatabase.addTask} 写入 RDB。
          \item \code{FocusStore.refresh} 重新读取并触发页面刷新。
        \end{enumerate}
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{后端接口覆盖}
  \small
  \begin{tabularx}{\linewidth}{p{2.25cm}p{3.7cm}X}
    \toprule
    模块 & 接口 & 作用 \\
    \midrule
    用户 & \code{/api/user/login}、\code{/register} & 登录、注册、连通性检查，返回 JWT。 \\
    项目 & \code{/api/projects} & 课程项目增删改查和软删除。 \\
    任务 & \code{/api/tasks} & 任务分页、创建、更新、删除。 \\

  \vspace{0.4em}
  \begin{block}{鉴权方式}
    \small
    登录成功后移动端在 \code{Authorization: Bearer <token>} 中携带 JWT，后端拦截器校验后将用户标识写入请求上下文，业务接口不信任客户端传入的用户编号。
  \end{block}
\end{frame}

\begin{frame}{双向同步：先拉取，再上推}
  \begin{columns}[T,onlytextwidth]
    \column{0.46\textwidth}
      \centering
      \includegraphics[height=4.7cm]{app-sync.jpeg}\\[-0.2em]
      {\scriptsize\color{muted}云同步页：地址、登录和同步操作}
    \column{0.50\textwidth}
      \begin{block}{客户端流程}
        \small
        \begin{enumerate}
          \item 读取 \code{lastServerTime} 作为增量游标。
          \item 调用 \code{pullFromServer} 拉取服务端变更。
          \item 按 \code{updatedAt} 合并到本地 RDB。
          \item 查询本地 \code{dirty=1} 数据并调用 \code{pushToServer}。
          \item 上推成功后清除 dirty，保存新的服务端时间。
        \end{enumerate}
      \end{block}
      \tagbox{softBlue}{serverTime 游标}
      \tagbox{softGreen}{dirty 增量}
      \tagbox{softAmber}{updatedAt LWW}
  \end{columns}
\end{frame}

\begin{frame}{同步一致性：避免重复和“删除后恢复”}
  \begin{center}
    \resizebox{0.95\linewidth}{!}{%
      \begin{tikzpicture}[
        node distance=0.42cm and 0.65cm,
        box/.style={rectangle, rounded corners=3pt, draw=lineGray, fill=softGray, minimum width=2.85cm, minimum height=0.68cm, align=center, font=\scriptsize},
        arrow/.style={-{Latex[length=2mm]}, thick, draw=accentLine}
      ]
        \node[box, fill=softBlue] (local) {本地新增/修改/删除\\dirty=1};
        \node[box, right=of local, fill=softGreen] (push) {Push 请求\\tasks/projects/pomodoros};
        \node[box, right=of push, fill=softAmber] (server) {后端事务写入\\@Transactional};
        \node[box, below=of server, fill=softRed] (judge) {查 id 或 clientRequestId\\比较 updatedAt};
        \node[box, left=of judge, fill=softGreen] (mark) {上推成功\\清除 dirty};
        \node[box, left=of mark, fill=softBlue] (pull) {下次 Pull\\携带删除墓碑};
        \draw[arrow] (local) -- (push);
        \draw[arrow] (push) -- (server);
        \draw[arrow] (server) -- (judge);
        \draw[arrow] (judge) -- (mark);
        \draw[arrow] (mark) -- (pull);
      \end{tikzpicture}
    }
  \end{center}

  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{重复提交：}任务和番茄记录带 \code{clientRequestId}，服务端先查同一请求标识再决定插入或更新。
    \item \textbf{冲突处理：}客户端和服务端都比较 \code{updatedAt}，保留不旧于已有版本的数据。
    \item \textbf{删除传播：}项目和任务保留 \code{isDeleted} tombstone，同步时继续上推删除语义。
  \end{itemize}
\end{frame}
